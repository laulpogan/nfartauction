      game: GameState,
      passerIdx: number,  // position of the player who is passing
    ): { updatedGame: GameState; auctioneerTakesFree: boolean } {
      if (!game.auction || game.auction.status !== 'waiting_second') throw new Error('No second-card phase active')
      const { auction } = game
      const originalAuctioneerIdx = auction.cards[0] // we need original — store it
      // waitingSecondCardIdx advances clockwise

      if (nextIdx === auction.auctioneerIdx) {

        // Remove the card from auction, advance currentPlayerIdx
        return {
          updatedGame: { ...game, auction: null, currentPlayerIdx: (game.currentPlayerIdx + 1) % game.players.length },
          auctioneerTakesFree: true,

      }
      return {

        auctioneerTakesFree: false,
      }
    }

       function derivePublicState(game: GameState): PublicGameState {
         const { deck: _deck, auction, ...rest } = game
         let publicAuction: PublicAuctionState | null = null
         if (auction) {
           if (auction.status !== 'completed') {
             const sealedBidsPresence: Record<number, true> = {}
             for (const idx of Object.keys(auction.sealedBids)) {
               sealedBidsPresence[Number(idx)] = true
             }
             publicAuction = { ...auction, sealedBids: sealedBidsPresence }
           } else {
             // Auction completed — safe to reveal bids; cast to presence type for type compat
             const sealedBidsPresence: Record<number, true> = {}
             for (const [idx, _amt] of Object.entries(auction.sealedBids)) {
               sealedBidsPresence[Number(idx)] = true
             }
             publicAuction = { ...auction, sealedBids: sealedBidsPresence }
           }
         }
         return { ...rest, deck: [] as never[], auction: publicAuction }
       }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→server WebSocket | All player actions arrive here; no authentication; identity = connection id |
| server→client broadcast | All clients receive every GAME_STATE message; public projection must hide private fields |
| server storage (Durable Object) | Persisted ServerState is server-internal; safe from client access |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-1-01 | Information Disclosure | `broadcastState()` — sealedBids in GAME_STATE | mitigate | Replace broadcastState with broadcastStateSecure using derivePublicState; sealedBids values stripped until auction.status === 'completed' |
| T-1-02 | Information Disclosure | `broadcastState()` — deck in GAME_STATE | mitigate | derivePublicState always sets deck: [] in public projection; remaining cards never sent to clients |
| T-1-03 | Tampering | PLAY_SECOND_CARD — no turn enforcement | mitigate | Server checks `auction.waitingSecondCardIdx === session.position` before accepting message |
| T-1-04 | Elevation of Privilege | JOIN message — isHost trusted from client | mitigate | Server ignores msg.isHost; first connection = host, all subsequent = non-host |
| T-1-05 | Tampering | Player name — uncapped string in JOIN | mitigate | Handled by Zod schema in Plan 02 (ENG-05); Plan 01 is a prerequisite that exposes the surface |
| T-1-06 | Denial of Service | Very long player names bloating all broadcasts | transfer | Mitigated by Zod validation in Plan 02; acceptable risk before Plan 02 completes |
</threat_model>
