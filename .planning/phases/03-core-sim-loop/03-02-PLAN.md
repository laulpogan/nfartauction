       private advancingFromSimDay = false

       private async advanceFromSimDay(_reason: 'all_submitted' | 'timeout') {
         if (!this.state) return
         if (this.advancingFromSimDay) return
         this.advancingFromSimDay = true
         try {
           this.clearSimDayTimeout()

           const sim = this.state.sim
           const players = this.state.game.players

           // Resolve each player's slots in position order
           const updatedPlayerSimMap: Record<string, PlayerSimState> = {}
           const updatedPlayers = players.map(p => {
             const ps = this.state!.playerSim[p.sessionId]
             if (!ps) return p
             const { updatedPlayerSim, updatedPlayerMoney } = resolveSlots(
               ps,
               ps.scheduledSlots ?? [],
               sim,
               p,
             )
             updatedPlayerSimMap[p.sessionId] = { ...updatedPlayerSim, scheduledSlots: [] }
             // Mirror coolness onto the public player so opponents see it
             return { ...p, money: updatedPlayerMoney, coolness: updatedPlayerSim.coolness }
           })

           // Advance global sim state (drift = 0 for now; deterministic in v1)
           const { updatedSim } = advanceDay(sim, Object.values(updatedPlayerSimMap))
           this.state.sim = updatedSim

           this.state.playerSim = { ...this.state.playerSim, ...updatedPlayerSimMap }

           // Transition phase: sim_day → auction_round
           // Use current game.round (engine round counter) as the auction roundNumber
           const nextPhase: GamePhase = { type: 'auction_round', roundNumber: this.state.game.round }
           this.state.game = {
             ...this.state.game,
             players: updatedPlayers,
             phase: nextPhase,
             sim: updatedSim,
           }

           // Sync sessions money to match
           this.state = syncSessions(this.state)

           await this.persist()
           this.broadcastStateSecure()
           this.broadcastSimStatePrivate()
         } finally {
           this.advancingFromSimDay = false
         }
       }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→server SUBMIT_SLOTS | Untrusted client sends arbitrary slot arrays. Zod is the gate. |
| server→client GAME_STATE broadcast | Must not contain ANY playerSim entries; only state.sim (public) flows through. |
| server→client YOUR_SIM_STATE | Sent per-connection only; sender connection ID looked up against state.playerSim map. |
| timer race | Last-submission and timeout firing simultaneously could double-resolve. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-06 | Tampering | SUBMIT_SLOTS payload | mitigate | Zod discriminated-union extension validates slot count ≤20, slot.type ∈ 6 enum values, neighborhood ∈ 5 enum values OR null, id length 1-64 |
| T-3-07 | Spoofing | SUBMIT_SLOTS sender | mitigate | sender.id (Party.Connection.id) is the sessionId; lookup against state.sessions; reject if not present |
| T-3-08 | Information Disclosure | playerSim leak | mitigate | derivePublicState's Omit list does not include `sim` (public is fine) but PlayerSimState lives in `state.playerSim`, NEVER inside `state.game`, and is never serialized into the GAME_STATE broadcast |
| T-3-09 | Information Disclosure | YOUR_SIM_STATE wrong recipient | mitigate | broadcastSimStatePrivate iterates connections and looks up by conn.id (sessionId); only sends if entry exists for THAT connection |
| T-3-10 | Denial of Service | infinite SUBMIT_SLOTS spam | mitigate | Zod max(20) caps slot array; Phase 1 onMessage already drops malformed JSON silently; per-connection rate limiting deferred to a hardening pass |
| T-3-11 | Race Condition | timeout vs all-submitted | mitigate | this.advancingFromSimDay flag guards against re-entry; clearSimDayTimeout called inside the guarded section |
| T-3-12 | Tampering | money manipulation via SUBMIT_SLOTS | mitigate | sim-engine.resolveSlots clamps money floor at 0; SLOT_DEFINITIONS deltas are server-side constants; client cannot specify deltas, only slot type |
